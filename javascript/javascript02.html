05. 배열 : 데이터 불러오기 : for()문

for문을 사용하여 배열 안에 들어있는 데이터들을 간편하게 불러올 수 있습니다. 

{
    const arr = [100, 200, 300, 400, 500, 600, 700, 800, 900];

    console.log("05. 배열 : 데이터 불러오기 : for()문");

    for (let i = 0; i > arr.length; i++) { // 초깃값; 조건식; 증감값
        // 실행문 
        console.log(arr[i]);
    }

}

06. 배열 : 데이터 불러오기 : forEach()문

forEach문을 사용하여 배열 안에 들어있는 데이터들을 간편하게 불러올 수 있습니다. 

{
    const arr = [100, 200, 300, 400, 500, 600, 700, 800, 900];

    console.log("06. 배열 : 데이터 불러오기 : forEach()문");
    arr.forEach(function (element) {
        console.log(element);
    })

    // 02 index = 번호
    arr.forEach(function (element, index) {
        console.log(index);
    })

    //03 Array = 원본 배열 데이터
    arr.forEach(function (element, index, array) {
        console.log(array);
    })

    // 화살표
    arr.forEach((e) => {
        console.log(e);
    })

    // 02.
    arr.forEach(e => {
        console.log(e);
    })

    // 03
    arr.forEach(e => console.log(e));
}

09. 배열 : 데이터 불러오기 : map()

map()문을 사용하여 배열 안에 들어있는 데이터들을 간편하게 불러올 수 있습니다. 

{
    const arr = [100, 200, 300, 400, 500, 600, 700, 800, 900];
    console.log("09. 배열 : 데이터 불러오기 : map()");

    arr.forEach(function (e) {
        console.log(e) // 100~900
    })

    // 01
    arr.map(function (element) {
        console.log(element); // [100] ~ [900]
    })

    // 02
    arr.map(function (element, index) {
        console.log(index);
    })

    // 03
    arr.map(function (element, index, array) {
        console.log(array);
    })

}
